[docs]def meaning_of_life_url() -> str:
    """
    Fetch the meaning of life from https://en.wikipedia.org/wiki/Main_Page.
    """
    url = "https://en.wikipedia.org/api/rest_v1/page/summary/Monty_Python's_The_Meaning_of_Life"  # noqa
    r = requests.get(url)
    r.raise_for_status()
    j = r.json()
    return j["extract"]
